let month = parseInt(prompt("Введіть номер місяця (1-12):")); let season; let monthName; switch (month) { case 1: monthName = "Січень"; case 2: monthName = "Лютий"; case 12: season = "Зима"; break; case 3: monthName = "Березень"; case 4: monthName = "Квітень"; case 5: season = "Весна"; break; case 6: monthName = "Червень"; case 7: monthName = "Липень"; case 8: season = "Літо"; break; case 9: monthName = "Вересень"; case 10: monthName = "Жовтень"; case 11: monthName = "Листопад"; season = "Осінь"; break; default: alert("Некоректний номер місяця."); } if (season) { alert(monthName + " - " + season); }
